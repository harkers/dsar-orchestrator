```python
"""Prompt-asset loader (§4.1 of durant-pipeline-hardening design).

The loader reads `prompts/<prompt_id>.md` assets with YAML frontmatter,
verifies a canonical-seal-sha256 over (metadata - seal field) + body,
optionally strips droppable blocks, and returns a frozen PromptAsset.
"""
from __future__ import annotations

import hashlib
from typing import Mapping

import yaml


def compute_seal(meta: Mapping, body: str) -> str:
    """Hash full canonical frontmatter + body together.

    `seal_sha256` (if present in meta) is excluded — that field is the
    output, not an input. Canonicalisation via yaml.safe_dump with
    sorted keys handles arbitrary YAML types without TypeError.
    """
    sealed_meta = {k: v for k, v in meta.items() if k != "seal_sha256"}
    canonical = yaml.safe_dump(sealed_meta, sort_keys=True,
                               default_flow_style=False)
    return hashlib.sha256(
        (canonical + "\n---\n" + body).encode("utf-8")
    ).hexdigest()
```

```python
import functools
from dataclasses import dataclass
from pathlib import Path
from typing import Optional

_PROMPTS_DIR = Path(__file__).parent / "prompts"


class PromptIntegrityError(Exception):
    """Raised when a prompt asset's seal doesn't match the computed value,
    when the asset is missing, or when the file violates format rules
    (CRLF, malformed frontmatter, etc.).
    """


@dataclass(frozen=True)
class PromptAsset:
    prompt_id: str
    version: str
    canonical_seal_sha256: str
    effective_sha256: str
    applied_strips: tuple[str, ...]
    body: str


def _read_asset_file(prompt_id: str) -> tuple[dict, str]:
    """Read prompts/<prompt_id>.md → (metadata dict, body string).

    Raises PromptIntegrityError on:
    - missing file
    - CR character in body (LF-only requirement)
    - malformed YAML frontmatter
    """
    path = _PROMPTS_DIR / f"{prompt_id}.md"
    if not path.is_file():
        raise PromptIntegrityError(f"prompt asset not found: {path}")
    raw = path.read_bytes()
    if b"\r" in raw:
        raise PromptIntegrityError(
            f"prompt asset {path} contains CR characters (LF-only required)")
    text = raw.decode("utf-8")
    if not text.startswith("---\n"):
        raise PromptIntegrityError(
            f"prompt asset {path} missing YAML frontmatter (no leading ---)")
    end = text.find("\n---\n", 4)
    if end < 0:
        raise PromptIntegrityError(
            f"prompt asset {path} frontmatter not terminated by ---")
    fm_text = text[4:end]
    body = text[end + len("\n---\n"):]
    try:
        meta = yaml.safe_load(fm_text)
    except yaml.YAMLError as e:
        raise PromptIntegrityError(
            f"prompt asset {path} frontmatter unparseable: {e}") from e
    if not isinstance(meta, dict):
        raise PromptIntegrityError(
            f"prompt asset {path} frontmatter is not a mapping")
    return meta, body


class PromptLoader:
    @classmethod
    @functools.lru_cache(maxsize=64)
    def load(cls, prompt_id: str, *,
             strip_sections: tuple[str, ...] = ()) -> PromptAsset:
        meta, body = _read_asset_file(prompt_id)
        expected_seal = meta.get("seal_sha256")
        if not isinstance(expected_seal, str):
            raise PromptIntegrityError(
                f"prompt asset {prompt_id} frontmatter missing seal_sha256")
        actual_seal = compute_seal(meta, body)
        if actual_seal != expected_seal:
            raise PromptIntegrityError(
                f"prompt asset {prompt_id} seal mismatch\n"
                f"  expected: {expected_seal}\n"
                f"  actual:   {actual_seal}")
        if strip_sections:
            raise NotImplementedError("strip_sections — Task 4")
        effective_sha256 = hashlib.sha256(body.encode("utf-8")).hexdigest()
        return PromptAsset(
            prompt_id=str(meta.get("prompt_id", prompt_id)),
            version=str(meta.get("version", "")),
            canonical_seal_sha256=actual_seal,
            effective_sha256=effective_sha256,
            applied_strips=(),
            body=body,
        )

    @classmethod
    def _cache_clear(cls) -> None:
        """Test helper: clear lru_cache so monkeypatched _PROMPTS_DIR takes effect."""
        cls.load.cache_clear()
```

```python
import os


def _read_asset_file_at(path: Path) -> tuple[dict, str]:
    """Like _read_asset_file but takes a direct path (for sign CLI)."""
    if not path.is_file():
        raise PromptIntegrityError(f"asset not found: {path}")
    raw = path.read_bytes()
    # Normalise CRLF→LF only during read for sign; the sign output is LF-only.
    text = raw.decode("utf-8").replace("\r\n", "\n").replace("\r", "\n")
    if not text.startswith("---\n"):
        raise PromptIntegrityError(f"{path}: no leading ---")
    end = text.find("\n---\n", 4)
    if end < 0:
        raise PromptIntegrityError(f"{path}: frontmatter not terminated")
    fm_text = text[4:end]
    body = text[end + len("\n---\n"):]
    try:
        meta = yaml.safe_load(fm_text)
    except yaml.YAMLError as e:
        raise PromptIntegrityError(f"{path}: bad YAML: {e}") from e
    if not isinstance(meta, dict):
        raise PromptIntegrityError(f"{path}: frontmatter not mapping")
    return meta, body


def sign_asset(path: Path) -> str:
    """Compute the seal for `path` and rewrite the file atomically.

    Converts CRLF/CR to LF on input (quiet fix during signing). Uses
    ruamel.yaml to preserve frontmatter key order and round-trip fidelity.
    Returns the new seal hex.
    """
    from ruamel.yaml import YAML

    yaml_rt = YAML()
    yaml_rt.default_flow_style = False
    yaml_rt.preserve_quotes = True

    meta_dict, body = _read_asset_file_at(path)
    # Recompute seal over canonical metadata (sans seal_sha256) + body.
    new_seal = compute_seal(meta_dict, body)
    meta_dict["seal_sha256"] = new_seal

    # Reconstruct file: ---\n<frontmatter>\n---\n<body>
    from io import StringIO
    fm_buf = StringIO()
    yaml_rt.dump(meta_dict, fm_buf)
    fm_text = fm_buf.getvalue()
    new_text = f"---\n{fm_text}---\n{body}"

    tmp_path = path.with_suffix(path.suffix + ".tmp")
    with open(tmp_path, "w", encoding="utf-8", newline="\n") as f:
        f.write(new_text)
        f.flush()
        os.fsync(f.fileno())
    os.replace(tmp_path, path)
    return new_seal
```

```python
import gzip
import json
from datetime import date


class RegistryError(Exception):
    pass


def build_registry() -> None:
    """Enumerate prompts/*.md, append new (version, seal) pairs to
    _registry.json, write archive files. Atomic per-file writes via
    tmp + fsync + os.replace.
    """
    registry_path = _PROMPTS_DIR / "_registry.json"
    archive_root = _PROMPTS_DIR / "_archive"
    archive_root.mkdir(exist_ok=True)
    if registry_path.exists():
        current = json.loads(registry_path.read_text(encoding="utf-8"))
    else:
        current = {}

    today = date.today().isoformat()
    updates: dict[str, list[dict]] = {k: list(v) for k, v in current.items()}

    for asset_path in sorted(_PROMPTS_DIR.glob("*.md")):
        meta, body = _read_asset_file_at(asset_path)
        prompt_id = meta["prompt_id"]
        version = meta["version"]
        seal = meta["seal_sha256"]
        existing = updates.get(prompt_id, [])
        match = next((e for e in existing if e["version"] == version), None)
        if match is None:
            # New version — append + archive.
            existing.append({"version": version, "seal_sha256": seal,
                              "archived_at": today})
            updates[prompt_id] = existing
            _write_archive(archive_root, asset_path, prompt_id, version)
        else:
            if match["seal_sha256"] != seal:
                raise RegistryError(
                    f"{prompt_id} v{version}: seal changed from "
                    f"{match['seal_sha256']} to {seal}; version not bumped")
            # else: identical entry — idempotent; no archive overwrite.

    _atomic_write_json(registry_path, updates)


def _write_archive(archive_root: Path, asset_path: Path,
                   prompt_id: str, version: str) -> None:
    out_dir = archive_root / prompt_id
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / f"{version}.md.gz"
    tmp_path = out_path.with_suffix(".md.gz.tmp")
    raw = asset_path.read_bytes()
    with open(tmp_path, "wb") as f:
        gz = gzip.GzipFile(filename="", mode="wb", fileobj=f, mtime=0,
                            compresslevel=6)
        gz.write(raw)
        gz.close()
        f.flush()
        os.fsync(f.fileno())
    os.replace(tmp_path, out_path)


def _atomic_write_json(path: Path, data) -> None:
    tmp_path = path.with_suffix(".json.tmp")
    with open(tmp_path, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, sort_keys=True)
        f.write("\n")
        f.flush()
        os.fsync(f.fileno())
    os.replace(tmp_path, path)
```

```python
"""Smarter truncation (§4.3 of durant-pipeline-hardening design).

Default head_tail mode preserves both ends of long documents — Durant
signals cluster at the start (sender/subject metadata) but also appear
in the tail (signatures, performance discussions appended to threads).
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Optional

_MARKER_FMT = "\n\n[... %d characters elided ...]\n\n"
_MARKER_FMT_MIN_LEN = len(_MARKER_FMT % 0)
_MIN_AVAIL = 200
_MAX_CONVERGE_ITERS = 12


@dataclass(frozen=True)
class TruncationResult:
    """
    elided_start / elided_end refer to indices into the ORIGINAL text:
    text_original[elided_start:elided_end] is exactly the portion NOT
    present in `truncated`. When no truncation, both == original_char_count.
    """
    truncated: str
    mode: str
    original_char_count: int
    truncated_char_count: int
    elided_start: int
    elided_end: int


def truncate(text: str, max_chars: int, *,
             mode: str = "head_tail",
             head_ratio: float = 0.75,
             display_original_len: Optional[int] = None) -> TruncationResult:
    if not 0.5 <= head_ratio <= 0.95:
        raise ValueError(f"head_ratio out of range [0.5, 0.95]: {head_ratio}")
    orig = len(text)
    orig_for_marker = display_original_len if display_original_len is not None else orig
    if mode == "none":
        if orig > max_chars:
            raise ValueError(
                f"mode=none but text exceeds cap: {orig} > {max_chars}")
        return TruncationResult(text, "none", orig, orig, orig, orig)
    if orig <= max_chars:
        return TruncationResult(text, "none", orig, orig, orig, orig)
    if mode == "head_tail":
        head_n, tail_n, marker = _converge_sizes(
            max_chars, head_ratio, orig_for_marker)
        body = text[:head_n] + marker + text[-tail_n:]
        return TruncationResult(
            truncated=body, mode="head_tail",
            original_char_count=orig,
            truncated_char_count=len(body),
            elided_start=head_n,
            elided_end=orig - tail_n,
        )
    if mode == "structure_aware":
        raise NotImplementedError("structure_aware — Task 13")
    raise ValueError(f"unknown truncation mode: {mode}")


def _converge_sizes(max_chars: int, head_ratio: float,
                    orig_for_marker: int) -> tuple[int, int, str]:
    """Find (head_n, tail_n, marker) such that head_n + len(marker) + tail_n
    == max_chars exactly. Iterates because marker length depends on the
    elided count (number of digits varies).
    """
    marker_len = _MARKER_FMT_MIN_LEN
    for _ in range(_MAX_CONVERGE_ITERS):
        avail = max_chars - marker_len
        if avail < _MIN_AVAIL:
            raise ValueError(
                f"max_chars={max_chars} too small for head+tail with marker "
                f"(need ≥ {marker_len + _MIN_AVAIL})")
        head_n = int(avail * head_ratio)
        tail_n = avail - head_n
        elided_for_display = max(0, orig_for_marker - head_n - tail_n)
        marker = _MARKER_FMT % elided_for_display
        if len(marker) == marker_len:
            assert head_n + len(marker) + tail_n == max_chars
            return head_n, tail_n, marker
        marker_len = len(marker)
    raise AssertionError(
        f"_converge_sizes failed in {_MAX_CONVERGE_ITERS} iterations; bug")
```

```python
def count_subject_mentions_in_elided(text: str,
                                      result: TruncationResult,
                                      data_subject: dict) -> int:
    """Count case-insensitive substring matches of subject identifiers
    in the elided portion of `text`. AUDIT-ONLY — never injected into
    the LLM prompt (per §4.3 of the spec)."""
    if result.elided_start >= result.elided_end:
        return 0
    elided = text[result.elided_start:result.elided_end].lower()
    identifiers = [
        data_subject.get("full_name", "") or "",
        data_subject.get("email", "") or "",
    ]
    additional = data_subject.get("additional_emails", []) or []
    identifiers.extend(additional)
    total = 0
    for ident in identifiers:
        ident = (ident or "").strip().lower()
        if len(ident) < 3:
            continue
        total += elided.count(ident)
    return total
```
